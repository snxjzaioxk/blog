    int huor1,minute1;

    scanf("%d %d",&huor1,&minute1);//接收第一个时间

    if (diff < 0) {
        diff += 24 * 60; //如果时间差为负数，说明跨天了，增加24小时的分钟数
    }
    int diff_huor = diff / 60; //计算小时
    int diff_minute = diff % 60; //计算分钟
    printf("时间差为：%d小时 %d分钟\n", diff_huor, diff_minute); //输出时间差

}
